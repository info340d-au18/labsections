```jsx
describe('<App /> component', () => {
    it('works', () => {
        const wrapper = mount(<Work work={'work'}/>);
        const work = jest.spyOn(wrapper.instance(), 'work');
        expect(work).toHaveBeenCalledTimes(9999999);
        wrapper.unmount();
    })
});
```
